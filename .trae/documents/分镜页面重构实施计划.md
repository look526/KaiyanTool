# 分镜页面重构实施计划

## 1. 项目概述

### 1.1 项目目标
重构分镜页面，实现剧集 - 场景 - 分镜的三层结构，提供 AI 辅助生成、@提及功能、视频生成备选等核心功能。

### 1.2 项目周期
总工期：13 个工作日（原 11 天 + 补充功能 2 天）

### 1.3 项目团队
- 后端开发：1 人
- 前端开发：1 人
- 测试：1 人

---

#### Day 2：后端 API 实现
**任务：**

**详细步骤：**

   - 创建 `apps/api/src/controllers/episode.controller.ts`

     - `getEpisodes` - 获取剧集列表
     - `getEpisode` - 获取剧集详情

   - 创建 `apps/api/src/controllers/shot-alternative.controller.ts`
   - 实现方法：
     - `getAlternatives` - 获取备选视频列表
     - `createAlternative` - 创建备选视频
     - `setRecommended` - 设为推荐

4. **ShotDraft 控制器**
   - 创建 `apps/api/src/controllers/shot-draft.controller.ts`
   - 实现方法：
     - `getDrafts` - 获取草稿列表
     - `saveDraft` - 保存草稿
     - `updateDraft` - 更新草稿
     - `deleteDraft` - 删除草稿

5. **Mention 控制器**
   - 创建 `apps/api/src/controllers/mention.controller.ts`
   - 实现方法：
     - `getMentions` - 获取可提及的资源

6. **路由定义**
   - 创建 `apps/api/src/routes/episode.routes.ts`
   - 创建 `apps/api/src/routes/scene.routes.ts`
   - 创建 `apps/api/src/routes/shot-alternative.routes.ts`

```typescript
const MentionInput = ({ value, onChange, projectId }) => {
  const [mentions, setMentions] = useState<MentionItem[]>([]);
  const [showSuggestions, setShowSuggestions] = useState(false);
  const [query, setQuery] = useState('');
  const [cursorPosition, setCursorPosition] = useState(0);
  
  // 防抖搜索
  const debouncedSearch = useMemo(
    () => debounce(async (q: string) => {
      const data = await apiClient.getMentions(projectId, q);
      setMentions(data);
    }, 300),
    [projectId]
  );
  
  // 监听输入
  const handleInputChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    const newValue = e.target.value;
    onChange(newValue);
    setCursorPosition(e.target.selectionStart);
    
    // 检测@符号
    const atPosition = newValue.lastIndexOf('@', cursorPosition);
    if (atPosition !== -1) {
      const queryText = newValue.slice(atPosition + 1, cursorPosition);
      setQuery(queryText);
      setShowSuggestions(true);
      debouncedSearch(queryText);
    } else {
      setShowSuggestions(false);
    }
  };
  
  // 选择建议
  const handleSelect = (item: MentionItem) => {
    const beforeAt = value.slice(0, cursorPosition - query.length - 1);
    const afterCursor = value.slice(cursorPosition);
    const newValue = `${beforeAt}@${item.name} ${afterCursor}`;
    onChange(newValue);
    setShowSuggestions(false);
  };
  
  // ...
};
```

```typescript
async getMentions(req: Request, res: Response): Promise<void> {
  const { projectId } = req.params;
  const { q } = req.query;
  
  // 查询角色
  const characters = await prisma.character.findMany({
    where: {
      project_id: projectId,
      name: { contains: q as string, mode: 'insensitive' }
    },
    take: 5
  });
  
  // 查询物品
  const items = await prisma.item.findMany({
    where: {
      project_id: projectId,
      name: { contains: q as string, mode: 'insensitive' }
    },
    take: 5
  });
  
  // 查询场景
  const scenes = await prisma.scene.findMany({
    where: {
      project_id: projectId,
      OR: [
        { location: { contains: q as string, mode: 'insensitive' } },
        { description: { contains: q as string, mode: 'insensitive' } }
      ]
    },
    take: 5
  });
  
  // 查询素材
  const assets = await prisma.asset.findMany({
    where: {
      project_id: projectId,
      type: 'image',
      name: { contains: q as string, mode: 'insensitive' }
    },
    take: 5
  });
  
  const mentions = [
    ...characters.map(c => ({ id: c.id, type: 'character', name: c.name })),
    ...items.map(i => ({ id: i.id, type: 'item', name: i.name })),
    ...scenes.map(s => ({ id: s.id, type: 'scene', name: s.location })),
    ...assets.map(a => ({ id: a.id, type: 'asset', name: a.name || 'Untitled' }))
  ];
  
  res.json(mentions);
}
```

```typescript
async generateShotVideo(req: Request, res: Response): Promise<void> {
  const { id } = req.params;
  const { provider_id, model, prompt, aspect_ratio, resolution } = req.body;
  
  // 1. 获取分镜
  const shot = await prisma.shot.findUnique({
    where: { id },
    include: { Scene: true, Character: true }
  });
  
  if (!shot) {
    res.status(404).json({ error: 'Shot not found' });
    return;
  }
  
  // 2. 获取 AI 提供商
  const provider = await prisma.aIProvider.findUnique({
    where: { id: provider_id },
    include: { models: true }
  });
  
  if (!provider) {
    res.status(404).json({ error: 'Provider not found' });
    return;
  }
  
  // 3. 调用 AI 生成
  const startTime = Date.now();
  const videoUrl = await videoGenerationService.generate({
    provider,
    model,
    prompt,
    aspect_ratio,
    resolution
  });
  const generationTime = Date.now() - startTime;
  
  // 4. 更新分镜状态
  await prisma.shot.update({
    where: { id },
    data: { 
      video_url: videoUrl,
      status: 'completed'
    }
  });
  
  res.json({ 
    success: true, 
    video_url: videoUrl,
    generation_time: generationTime
  });
}

// 批量生成 API
async batchGenerateShots(req: Request, res: Response): Promise<void> {
  const { episodeId } = req.params;
  const { shot_ids, provider_id, model, aspect_ratio, resolution } = req.body;
  
  let successful = 0;
  let failed = 0;
  
  // 并行处理生成请求
  const promises = shot_ids.map(shotId => 
    videoGenerationService.generateShot(shotId, {
      provider_id,
      model,
      aspect_ratio,
      resolution
    }).then(() => successful++)
      .catch(() => failed++)
  );
  
  await Promise.all(promises);
  
  res.json({ 
    success: true,
    successful,
    failed,
    total: shot_ids.length
  });
}
```
